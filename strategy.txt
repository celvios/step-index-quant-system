5.3. Step Acceleration

def step_velocity(prices, threshold=3):
    count = 0
    for i in range(1, len(prices)):
        if abs(prices[i] - prices[i-1]) == 0.1:
            count += 1
        else:
            count = 0
        if count >= threshold:
            return True
return False

Volatility-Adaptive TP Framework

def calculate_dynamic_tp(entry, sl, rr_target, volatility):
    base_distance = abs(entry - sl)
    
    # Volatility multiplier
    if volatility < 0.5: mult = 0.8
    elif volatility > 1.5: mult = 1.5
    else: mult = 1.0
    
    tp_distance = base_distance * rr_target * mult
    return entry + tp_distance * trade_direction
